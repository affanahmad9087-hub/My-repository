    <button class="toggle-btn" onclick="toggleWeather()">SWAP WEATHER</button>

    <script>
        function toggleWeather() {
            const body = document.getElementById('mainBody');
            body.classList.toggle('stormy');
            
            // Just some text changing logic
            const desc = document.querySelector('.desc');
            const temp = document.querySelector('.temp');
            if(body.classList.contains('stormy')) {
                desc.innerText = "Electric Storm";
                temp.innerText = "14°C";
            } else {
                desc.innerText = "Clear Skies";
                temp.innerText = "28°C";
            }
        }
    </script>
